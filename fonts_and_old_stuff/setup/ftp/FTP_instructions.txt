```bash
# Install the necessary packages and start the firewall
sudo pacman -Syu vsftpd firewalld
```

```bash
# Start the firewall service
sudo systemctl enable --now firewalld
```

```bash
# Add your user to the userlist (replace your_username with your real username)
echo "your_username" | sudo tee /etc/vsftpd.userlist
```

```bash
# Open the config file in a text editor
sudo nvim /etc/vsftpd.conf
```

Inside the editor: **delete all existing text** and paste exactly the following lines:

```conf
anonymous_enable=NO
local_enable=YES
write_enable=YES
chroot_local_user=YES
allow_writeable_chroot=YES
local_root=/mnt/ramdisk/
userlist_enable=YES
userlist_file=/etc/vsftpd.userlist
userlist_deny=NO
xferlog_enable=YES
xferlog_std_format=YES
xferlog_file=/tmp/vsftpd.log
log_ftp_protocol=YES
listen=NO
listen_ipv6=YES
pam_service_name=vsftpd
pasv_enable=YES
pasv_min_port=40000
pasv_max_port=40100
ftpd_banner=Welcome to this Arch Linux FTP service.
use_sendfile=YES
connect_from_port_20=YES
```

Save and quit, then run:

```bash
sudo firewall-cmd --permanent --add-service=ftp
sudo firewall-cmd --permanent --add-port=40000-40100/tcp
sudo firewall-cmd --reload
sudo systemctl enable --now vsftpd.service
```

Done. (Make sure `/mnt/ramdisk/` exists and is mounted as tmpfs before starting the service.)
